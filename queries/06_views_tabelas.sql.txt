-- CREATE VIEW
create view despesa_total_cliente as
select
pay.customer_id,
sum(pay.amount) as total_amount,
cus.first_name,
cus.last_name
from payment pay
join customer cus on pay.customer_id = cus.customer_id
group by pay.customer_id, cus.first_name, cus.last_name;
